2.

[tex]\int\limits^4_1 {3\sqrt{x} } \, dx[/tex]

[tex]= 3 \int\limits {\sqrt{x} } \, dx[/tex]

[tex]= 3 \int\limits x^{\frac{1}{2} } \, dx[/tex]

[tex]= 3 . \frac{2x^{\frac{3}{2} } }{3}[/tex]

[tex]= 2x^{\frac{3}{2} }[/tex]

[tex]= 2x\sqrt{x}[/tex]

hitung nilai integral

[tex]= 2x\sqrt{x} I^4_1[/tex]

[tex]= 2. 4\sqrt{4} - 2.1\sqrt{1}[/tex]

[tex]= 16 - 2\\= 14[/tex]

3.

[tex]\int\limits^5_2 {\frac{4}{x^{2} } } \, dx[/tex]

[tex]= 4\int\limits {\frac{1}{x^{2} } } \, dx[/tex]

[tex]= 4 . (-\frac{1}{x} )[/tex]

[tex]=-\frac{4}{x}[/tex]

hitung nilai integral

[tex]=-\frac{4}{x} I^5_2[/tex]

[tex]=-\frac{4}{5} -(-\frac{4}{2} )[/tex]

[tex]=-\frac{4}{5} -(-2 )[/tex]

[tex]=-\frac{4}{5} -(-\frac{10}{5} )[/tex]

[tex]=\frac{6}{5}[/tex]

4.

[tex]\int\limits^1_0 {(2x+1)^{3} } \, dx[/tex]

misal 2x + 1 = u

[tex]=\int\limits {(2x+1)^{3} } \, dx[/tex]

[tex]=\int\limits {\frac{u^{3} }{2} } \, du[/tex]

[tex]=\frac{1}{2} \int\limits u^{3} \, du[/tex]

[tex]=\frac{1}{2} . \frac{u^{4} }{4}[/tex]

[tex]=\frac{1}{2} . \frac{(2x+1)^{4} }{4}[/tex]

[tex]= \frac{(2x+1)^{4} }{8}[/tex]

hitung nilai integral

[tex]= \frac{(2x+1)^{4} }{8} I^1_0[/tex]

[tex]= \frac{(2.1+1)^{4} }{8} - \frac{(2.0+1)^{4} }{8}[/tex]

[tex]= \frac{(3)^{4} }{8} - \frac{(1)^{4} }{8}[/tex]

[tex]=\frac{81}{8} -\frac{1}{8}[/tex]

[tex]=\frac{80}{8}\\=10[/tex]
