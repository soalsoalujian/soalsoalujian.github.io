8. soal UAS kls 12 IPA no 25, 26, dan 27


27. r = 6% = 0,06 per semester
M¹ = tabungan awal
M² = tabungan akhir
1 tahun = 2 semester
n = 20 tahun = 2 x 20 = 40 semester
M² = M¹ x ( 1 + r )^n
11.828.575,63 = M¹ x (1+0,06)^40
11.828.575,63 = M¹ x 1,06^40
M¹ = 11.828.575,63/1,06^40
= 11.828.575,63/10,28571794 = Rp. 1.150.000,oo

[tex] = (\frac{{2p}^{2}q {r}^{ - 3} }{4pq {r}^{ - 2} }) {}^{2} [/tex]

[tex] = ( \frac{ {2}^{2} {p}^{(2 \times 2)} {q}^{2} {r}^{ (- 3 \times 2)} }{ {4}^{2} {p}^{2} {q}^{2} {r}^{( - 2 \times 2)} } )[/tex]

[tex] = ( \frac{4 {p}^{4 } {q}^{2} {r}^{ - 6} }{16 {p}^{2} {q}^{2} {r}^{ - 4} } )[/tex]

[tex] = \frac{ {p}^{(4 - 2)} {q}^{(2 - 2)} {r}^{( - 6 - ( - 4)} }{4} [/tex]

[tex] = \frac{ {p}^{2} {r}^{( - 6 + 4)} }{4} [/tex]

[tex] = \frac{ {p}^{2} {r}^{ - 2} }{4} [/tex]
